108    def merge_camera_data(
109        self,
110        *datasets: list[dict[str, Any]],
111        match_by: str = "coordinates",
112        threshold: float = 0.1,
113        check_id: bool = False,
114        check_url: bool = False,
115    ) -> list[dict[str, Any]]:
116        """
117        Merges one or more datasets of highway cameras, removing duplicates per highway.
118        Datasets should be ordered by priority (highest priority first).
119
120        Args:
121            *datasets (list[dict[str, Any]]): Variable number of datasets to merge.
122            match_by (str, optional): The method to match duplicates, either "coordinates"
123                or "km_point". Defaults to "coordinates".
124            threshold (float, optional): The threshold distance for spatial matching (in km)
125                or km point matching. Defaults to 0.1.
126            check_id (bool, optional): Whether to check for duplicate camera IDs. Defaults to False.
127            check_url (bool, optional): Whether to deduplicate by exact URL matching. Defaults to False.
128
129        Raises:
130            ValueError: If `match_by` is not 'coordinates' or 'km_point'.
131
132        Returns:
133            list[dict[str, Any]]: The merged and deduplicated list of highway dictionary entries.
134        """
135        if match_by not in {"coordinates", "km_point"}:
136            raise ValueError("match_by must be 'coordinates' or 'km_point'")
137
138        def _coords(cam: dict[str, Any]) -> tuple[float, float] | None:
139            """Extracts (X, Y) coordinates from a camera."""
140            c = cam.get("coords") or {}
141            x, y = c.get("X"), c.get("Y")
142            if x is not None and y is not None:
143                return (x, y)
144            return None
145
146        def _spatial_match(cam1: dict[str, Any], cam2: dict[str, Any]) -> bool:
147            """Checks if two cameras are geographically close or have a similar km point."""
148            if match_by == "coordinates":
149                p1, p2 = _coords(cam1), _coords(cam2)
150                if p1 is None or p2 is None:
151                    return False
152                return haversine_km(p1[1], p1[0], p2[1], p2[0]) <= threshold
153            km1 = cam1.get("camera_km_point")
154            km2 = cam2.get("camera_km_point")
155            if (km1 is not None and km2 is not None) and (abs(km1 - km2) <= threshold):
156                return True
157            return False
158
159        def _is_duplicate(cam: dict[str, Any], existing: dict[str, Any]) -> bool:
160            """Same-ID duplicate: both coords missing OR spatially close (coordinates mode only)."""
161            if _coords(cam) is None and _coords(existing) is None:
162                return True
163            return _spatial_match(cam, existing) if match_by == "coordinates" else False
164
165        # name -> list of cameras
166        merged: dict[str, list[dict[str, Any]]] = {}
167        countries: dict[str, str] = {}
168
169        for dataset in datasets:
170            if not dataset:
171                continue
172            for entry in dataset:
173                highway = entry.get("highway", {})
174                name = highway.get("name")
175                if not name:
176                    continue
177
178                countries.setdefault(name, highway.get("country", self.country))
179                target = merged.setdefault(name, [])
180                seen_urls = (
181                    {c.get("url") for c in target if c.get("url")}
182                    if check_url
183                    else set()
184                )
185                cameras_by_id = (
186                    {c["camera_id"]: c for c in target if c.get("camera_id")}
187                    if check_id
188                    else {}
189                )
190
191                for cam_in in highway.get("cameras", []):
192                    cam = copy.deepcopy(cam_in)
193                    url = cam.get("url")
194                    cam_id = cam.get("camera_id")
195
196                    # URL dedup
197                    if check_url and url and url in seen_urls:
198                        continue
199
200                    # ID-based dedup
201                    if check_id and cam_id:
202                        existing = cameras_by_id.get(cam_id)
203                        if existing:
204                            if _is_duplicate(cam, existing):
205                                continue
206                            # Rename to avoid ID collision
207                            i = 1
208                            new_id = f"{cam_id}_dup{i}"
209                            while new_id in cameras_by_id:
210                                i += 1
211                                new_id = f"{cam_id}_dup{i}"
212                            cam["camera_id"] = new_id
213                            cam_id = new_id
214
215                    # Spatial dedup (only when not using ID-based checks)
216                    if not check_id and any(_spatial_match(cam, c) for c in target):
217                        continue
218
219                    target.append(cam)
220                    if check_url and url:
221                        seen_urls.add(url)
222                    if cam_id:
223                        cameras_by_id[cam_id] = cam
224
225        return [
226            {
227                "highway": {
228                    "name": name,
229                    "country": countries[name],
230                    "cameras": sorted(
231                        cams, key=lambda c: c.get("camera_km_point", 0.0)
232                    ),
233                }
234            }
235            for name, cams in sorted(merged.items())
236        ]

238    async def get_parsed_data(
239        self,
240        output_file: str | Path | None = None,
241        output_folder: str | Path | None = None,
242    ) -> Any:
243        """
244        Orchestrates downloading and parsing data.
245
246        Args:
247            output_file (str | Path | None, optional): Specific file path to save output. Defaults to None.
248            output_folder (str | Path | None, optional): Folder to save output according to country format. Defaults to None.
249
250        Returns:
251            Any: The parsed camera data.
252        """
253        raw_data = None
254        if self.downloader:
255            raw_data = await self.downloader.get_data()
256
257        # Handle async/sync parse method
258        if inspect.iscoroutinefunction(self.parse):
259            parsed_data = await self.parse(raw_data)
260        else:
261            parsed_data = self.parse(raw_data)
262
263        if output_file:
264            save_json(parsed_data, output_file)
265        elif output_folder:
266            file_name = f"cameras_{self.country.lower()}{'_gov' if self.country in ['ES', 'UK'] else ''}.json"  # France(FR) and Italy(IT) handle saving independently
267            save_json(parsed_data, Path(output_folder) / file_name)
268
269        return parsed_data

119        Datasets should be ordered by priority (highest priority first).
120
121        Args:
122            *datasets (list[dict[str, Any]]): Variable number of datasets to merge.
123            match_by (str, optional): The method to match duplicates, either "coordinates"
124                or "km_point". Defaults to "coordinates".
125            threshold (float, optional): The threshold distance for spatial matching (in km)
126                or km point matching. Defaults to 0.1.
127            check_id (bool, optional): Whether to check for duplicate camera IDs. Defaults to False.
128            check_url (bool, optional): Whether to deduplicate by exact URL matching. Defaults to False.
129
130        Raises:
131            ValueError: If `match_by` is not 'coordinates' or 'km_point'.
132
133        Returns:
134            list[dict[str, Any]]: The merged and deduplicated list of highway dictionary entries.
135        """
136        if match_by not in {"coordinates", "km_point"}:
137            raise ValueError("match_by must be 'coordinates' or 'km_point'")
138
139        def _coords(cam: dict[str, Any]) -> tuple[float, float] | None:
140            """Extracts (X, Y) coordinates from a camera."""
141            c = cam.get("coords") or {}
142            x, y = c.get("X"), c.get("Y")
143            if x is not None and y is not None:
144                return (x, y)
145            return None
146
147        def _spatial_match(cam1: dict[str, Any], cam2: dict[str, Any]) -> bool:
148            """Checks if two cameras are geographically close or have a similar km point."""
149            if match_by == "coordinates":
150                p1, p2 = _coords(cam1), _coords(cam2)
151                if p1 is None or p2 is None:
152                    return False
153                return haversine_km(p1[1], p1[0], p2[1], p2[0]) <= threshold
154            km1 = cam1.get("camera_km_point")
155            km2 = cam2.get("camera_km_point")
156            if (km1 is not None and km2 is not None) and (abs(km1 - km2) <= threshold):
157                return True
158            return False
159
160        def _is_duplicate(cam: dict[str, Any], existing: dict[str, Any]) -> bool:
161            """Same-ID duplicate: both coords missing OR spatially close (coordinates mode only)."""
162            if _coords(cam) is None and _coords(existing) is None:
163                return True
164            return _spatial_match(cam, existing) if match_by == "coordinates" else False
165
166        # name -> list of cameras
167        merged: dict[str, list[dict[str, Any]]] = {}
168        countries: dict[str, str] = {}
169
170        for dataset in datasets:
171            if not dataset:
172                continue
173            for entry in dataset:
174                highway = entry.get("highway", {})
175                name = highway.get("name")
176                if not name:
177                    continue
178
179                countries.setdefault(name, highway.get("country", self.country))
180                target = merged.setdefault(name, [])
181                seen_urls = (
182                    {c.get("url") for c in target if c.get("url")}
183                    if check_url
184                    else set()
185                )
186                cameras_by_id = (
187                    {c["camera_id"]: c for c in target if c.get("camera_id")}
188                    if check_id
189                    else {}
190                )
191
192                for cam_in in highway.get("cameras", []):
193                    cam = copy.deepcopy(cam_in)
194                    url = cam.get("url")
195                    cam_id = cam.get("camera_id")
196
197                    # URL dedup
198                    if check_url and url and url in seen_urls:
199                        continue
200
201                    # ID-based dedup
202                    if check_id and cam_id:
203                        existing = cameras_by_id.get(cam_id)
204                        if existing:
205                            if _is_duplicate(cam, existing):
206                                continue
207                            # Rename to avoid ID collision
208                            i = 1
209                            new_id = f"{cam_id}_dup{i}"
210                            while new_id in cameras_by_id:
211                                i += 1
212                                new_id = f"{cam_id}_dup{i}"
213                            cam["camera_id"] = new_id
214                            cam_id = new_id
215
216                    # Spatial dedup (only when not using ID-based checks)
217                    if not check_id and any(_spatial_match(cam, c) for c in target):
218                        continue
219
220                    target.append(cam)
221                    if check_url and url:
222                        seen_urls.add(url)
223                    if cam_id:
224                        cameras_by_id[cam_id] = cam
225
226        return [
227            {
228                "highway": {
229                    "name": name,
230                    "country": countries[name],
231                    "cameras": sorted(
232                        cams, key=lambda c: c.get("camera_km_point", 0.0)
233                    ),
234                }
235            }
236            for name, cams in sorted(merged.items())
237        ]
