 33    def parse_autostrade_cameras(self, raw_data: str | bytes) -> list[dict[str, Any]]:
 34        """
 35        Parses JSON data for Autostrade cameras. (Most of the country cameras are here)
 36
 37        Args:
 38            raw_data (str | bytes): The raw JSON string or bytes containing Autostrade data.
 39
 40        Returns:
 41            list[dict[str, Any]]: A list of formatted highway camera dictionaries.
 42        """
 43        if not raw_data:
 44            return []
 45        try:
 46            data = load_json(raw_data)
 47        except Exception as e:
 48            print(f"Error parsing Autostrade JSON: {e}")
 49            return []
 50
 51        grouped_highways: dict[str, list[dict[str, Any]]] = defaultdict(list)
 52        webcams: list[dict[str, Any]] = data.get("webcams", [])
 53
 54        for cam in webcams:
 55            highway_name: str = cam.get("c_str", "Unknown")
 56            if highway_name == "A4":
 57                highway_name = "A04"
 58
 59            video_fragment: str = cam.get("frames", {}).get("V", {}).get("t_url", "")
 60            if not video_fragment:
 61                continue
 62
 63            full_url: str = f"{CAMERA_BASE_URL}{video_fragment}"
 64
 65            km_ini: float | None = cam.get("n_prg_km_ini")
 66            km_fin: float | None = cam.get("n_prg_km_fin")
 67
 68            direction = "*"
 69            if km_ini is not None and km_fin is not None:
 70                if km_ini < km_fin:
 71                    direction = "+"
 72                elif km_ini > km_fin:
 73                    direction = "-"
 74
 75            camera_entry = self.format_camera(
 76                camera_id=cam.get("c_tel"),
 77                camera_km_point=cam.get("n_prg_km", 0.0),
 78                camera_view=direction,
 79                camera_type="vid",
 80                coord_x=cam.get("n_crd_lon"),
 81                coord_y=cam.get("n_crd_lat"),
 82                url=full_url,
 83            )
 84            grouped_highways[highway_name].append(camera_entry)
 85
 86        return self.format_highway_output(grouped_highways)

294    async def parse(self, raw_data: dict[str, Any]) -> list[dict[str, Any]]:
295        """
296        Parses data from all Italian sources and merges them into a single list.
297
298        Args:
299            raw_data (dict[str, Any]): A dictionary mapping data sources to their raw content.
300
301        Returns:
302            list[dict[str, Any]]: The merged and deduplicated list of Italian highway cameras.
303        """
304        parsed_data = self.parse_autostrade_cameras(raw_data.get("autostrade", ""))
305
306        a22_parsed = self.parse_a22_cameras(raw_data.get("a22", ""))
307        if a22_parsed:
308            parsed_data.extend(a22_parsed)
309
310        a4_cameras: list[dict[str, Any]] = []
311
312       # As Italy has various concessionaries for the main road, we need to handle the government + 3 other sections.
313        a4_cameras.extend(self.parse_a4_abp(raw_data.get("a4_abp", "")))
314        a4_cameras.extend(self.parse_a4_cav(raw_data.get("a4_cav", "")))
315        a4_cameras.extend(self.parse_a4_satap(raw_data.get("a4_satap", "")))
316
317        if a4_cameras:
318            a4_entry = next(
319                (h for h in parsed_data if h["highway"]["name"] == "A04"), None
320            )
321            if a4_entry:
322                a4_entry["highway"]["cameras"].extend(a4_cameras)
323            else:
324                # This shouldn't happen, but just in case
325                parsed_data.append(
326                    {
327                        "highway": {
328                            "name": "A04",
329                            "country": self.country,
330                            "cameras": a4_cameras,
331                        }
332                    }
333                )
334
335        # De-duplicate cameras based on coordinates
336        parsed_data = self.merge_camera_data(
337            parsed_data,
338            match_by="coordinates",
339            threshold=0.015,
340            check_id=True,
341            check_url=True,
342        )
343
344        total_cameras = sum(len(h["highway"]["cameras"]) for h in parsed_data)
345        print(f"Successfully parsed {total_cameras} cameras.")
346        print(f"Grouped into {len(parsed_data)} highways")
347        return parsed_data
