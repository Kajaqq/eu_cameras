 28    @property
 29    def country(self) -> str:
 30        """
 31        Property that returns the country code.
 32
 33        Returns:
 34            str: The two-letter country code ('FR').
 35        """
 36        return "FR"

 38    @staticmethod
 39    def _extract_highway_name(text: str, camera_id: str | None = None) -> str:
 40        """
 41        Extracts a normalized highway name from raw text or falls back to a
 42        known mapping using the camera ID.
 43
 44        Args:
 45            text (str): The raw text description containing the highway name.
 46            camera_id (str | None, optional): The ID of the camera, used for fallback
 47                mapping if the text doesn't explicitly match the road regex. Defaults to None.
 48
 49        Returns:
 50            str: The normalized highway name (e.g., 'A-10', 'N-154') or 'Unknown'.
 51        """
 52        def _normalize_rn(prefix_:str):
 53            # France national highways 'Route Nationale' use RN or N interchangeably. We normalize it to more common 'N'
 54            return "N" if prefix == "RN" else prefix
 55
 56        if text:
 57            # Try to get the first apperance of a highway name pattern
 58            match = ROAD_REGEX.search(text)
 59            if match:
 60                prefix = match.group(1).upper()
 61                prefix = _normalize_rn(prefix)
 62                return f"{prefix}-{match.group(2)}"
 63
 64        # This is needed as France metadata sometimes doesn't include the specific name
 65        # The list has been manually checked based on coordinates
 66        if camera_id and camera_id in UNKNOWN_MAPPING:
 67            mapped_name = UNKNOWN_MAPPING[camera_id]
 68            match = ROAD_REGEX.search(mapped_name)
 69            if match:
 70                prefix = match.group(1).upper()
 71                prefix =  _normalize_rn(prefix)
 72                return f"{prefix}-{match.group(2)}"
 73            return mapped_name
 74
 75        return "Unknown"

 77    def parse_gov_cameras(self, gov_baguettes: str | bytes) -> list[dict[str, Any]]:
 78        """
 79        Parses raw GeoJSON data from the French government (Bison Futé).
 80
 81        Args:
 82            gov_baguettes (str | bytes): The raw GeoJSON string or bytes.
 83
 84        Returns:
 85            list[dict[str, Any]]: A list of formatted highway camera dictionaries.
 86        """
 87        def _km_point_get(full_label_) -> float:
 88            km_pt = 0.0
 89            pr_match = PR_REGEX.search(full_label_)
 90            if pr_match:
 91                km = int(pr_match.group(1))
 92                meters_str = pr_match.group(2)
 93                meters = int(meters_str) if meters_str else 0
 94                km_pt = km + (meters / 1000.0)
 95            return km_pt
 96
 97        try:
 98            raw_data = json.loads(gov_baguettes)
 99        except (json.JSONDecodeError, Exception) as e:
100            print(f"Error decoding Gov JSON: {e}")
101            return []
102
103        grouped_highways: dict[str, list[dict[str, Any]]] = defaultdict(list)
104        features: list[dict[str, Any]] = raw_data.get("features") or []
105        camera_sum = 0
106
107        for feature in features:
108            camera_sum += 1
109            props: dict[str, Any] = feature.get("properties") or {}
110            geometry: dict[str, Any] = feature.get("geometry") or {}
111            full_label: str = props.get("libelleCamera") or ""
112            camera_id: str = feature.get("id", "")
113            km_point = _km_point_get(full_label)
114            flux_type: str = props.get("typeFlux") or ""
115            cam_type = (
116                "vid"
117                if flux_type == "VIDEO"
118                else "img"
119                if flux_type == "IMAGE"
120                else "unknown"
121            )
122
123            coords_in: list[float] = geometry.get("coordinates") or []
124            lon, lat = (
125                convert_to_wgs84(coords_in[0], coords_in[1])
126                if len(coords_in) >= 2
127                else (0.0, 0.0)
128            )
129
130            highway_name = self._extract_highway_name(full_label, camera_id)
131            camera_entry = self.format_camera(
132                camera_id=camera_id,
133                camera_km_point=round(km_point, 3),
134                camera_view="*",
135                camera_type=cam_type,
136                coord_x=round(lon, 6),
137                coord_y=round(lat, 6),
138            )
139            grouped_highways[highway_name].append(camera_entry)
140
141        print(f"Succesfully parsed {camera_sum} Government cameras")
142        return self.format_highway_output(grouped_highways)

144    def parse_asfa_cameras(self, asfa_baguettes: str) -> list[dict[str, Any]]:
145        """
146        Parses raw Javascript array data from ASFA (Autoroutes.fr).
147
148        Args:
149            asfa_baguettes (str): The raw JavaScript code snippet containing camera data.
150
151        Returns:
152            list[dict[str, Any]]: A list of formatted highway camera dictionaries.
153        """
154        try:
155            data_string = asfa_baguettes.strip()
156            data_string = re.sub(r"^var\s+\w+\s*=\s*", "", data_string)
157            data_string = re.sub(r";\s*\w+\.\w+\(.*\);?$", "", data_string)
158            parsed_data: list[Any] = ast.literal_eval(data_string)
159        except Exception as e:
160            print(f"Error parsing ASFA data: {e}")
161            return []
162
163        grouped_highways: dict[str, list[dict[str, Any]]] = defaultdict(list)
164        camera_sum = 0
165        for item in parsed_data:
166            coords, _, _, description, metadata = item
167            camera_id: str = metadata.get("id")
168            camera_sum += 1
169
170            highway_name = self._extract_highway_name(description, camera_id)
171
172            # ASFA doesn't return km_point or camera direction
173            camera_entry = self.format_camera(
174                camera_id=camera_id,
175                camera_km_point=0.0,
176                camera_view="*",
177                camera_type="asfa_vid",
178                coord_x=float(coords[1]),
179                coord_y=float(coords[0]),
180            )
181
182            grouped_highways[highway_name].append(camera_entry)
183
184        print(f"Succesfully parsed {camera_sum} ASFA cameras")
185        return self.format_highway_output(grouped_highways)

187    async def parse(
188        self, raw_data: tuple[str, str | bytes]
189    ) -> tuple[list[dict[str, Any]], list[dict[str, Any]], list[dict[str, Any]]]:
190        """
191        Parses both ASFA and Government raw datasets and merges them.
192
193        Args:
194            raw_data (tuple[str, str | bytes]): A tuple containing the raw ASFA string
195                and raw Government GeoJSON string/bytes respectively.
196
197        Returns:
198            tuple[list[dict[str, Any]], list[dict[str, Any]], list[dict[str, Any]]]:
199                A tuple holding:
200                - The list of parsed Government highway cameras.
201                - The list of parsed ASFA highway cameras.
202                - The list of merged and deduplicated highway cameras.
203        """
204        asfa_raw, gov_raw = raw_data
205        gov_cameras = self.parse_gov_cameras(gov_raw) if gov_raw else []
206        asfa_cameras = self.parse_asfa_cameras(asfa_raw) if asfa_raw else []
207        merged_data = self.merge_camera_data(
208            gov_cameras, asfa_cameras, match_by="coordinates", threshold=0.20
209        )
210        print(f"Merged cameras grouped into {len(merged_data)} highways")
211        return gov_cameras, asfa_cameras, merged_data

214async def get_parsed_data(
215    output_file_gov: str | Path | None = None,
216    output_file_asfa: str | Path | None = None,
217    output_file_merged: str | Path | None = None,
218    output_folder: str | Path | None = None,
219) -> list[dict[str, Any]]:
220    """
221    Downloads, parses, and merges French camera data from multiple sources.
222
223    This function handles special intermediate JSON saving requirements
224    unique to the French datasets.
225
226    Args:
227        output_file_gov (str | Path | None, optional): File path for saving parsed Gov data.
228        output_file_asfa (str | Path | None, optional): File path for saving parsed ASFA data.
229        output_file_merged (str | Path | None, optional): File path for saving merged data.
230        output_folder (str | Path | None, optional): Folder to save raw files and defaults.
231
232    Returns:
233        list[dict[str, Any]]: The merged list of French highway camera data.
234    """
235    downloader = FranceDownloader()
236    asfa_raw, gov_raw = await downloader.get_data()
237    raw_data = asfa_raw, gov_raw
238    parser = FranceParser()
239
240    gov_cameras, asfa_cameras, merged_data = await parser.parse(raw_data)
241
242    if output_file_merged:
243        save_json(merged_data, output_file_merged)
244
245    # Allow saving camera sources separately
246    if output_file_gov and gov_cameras:
247        save_json(gov_cameras, output_file_gov)
248    if output_file_asfa and asfa_cameras:
249        save_json(asfa_cameras, output_file_asfa)
250
251    if output_folder:
252        # If output folder is specified, save all files
253        folder_path = Path(output_folder)
254        output_file_gov_name = "cameras_fr_gov.json"
255        output_file_asfa_name = "cameras_fr_asfa.json"
256        output_file_merged_name = "cameras_fr_merged.json"
257        save_json(asfa_raw, folder_path / output_file_asfa_name)
258        save_json(gov_raw, folder_path / output_file_gov_name)
259        save_json(merged_data, folder_path / output_file_merged_name)
260
261    return merged_data
