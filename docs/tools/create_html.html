 37    cameras: list[tuple[str, str, str, int, str]] = []
 38
 39    if camera_ids:
 40        camera_map: dict[str, tuple[dict[str, Any], str]] = {}
 41        for highway_item in json_data:
 42            hw_name = highway_item["highway"]["name"]
 43            for cam in highway_item["highway"]["cameras"]:
 44                camera_map[cam["camera_id"]] = (cam, hw_name)
 45
 46        camera_number = 1
 47
 48        for cid in camera_ids:
 49            if cid in camera_map:
 50                camera, highway_name = camera_map[cid]
 51

 53                    url = camera["url"]
 54                    media_type = "video"
 55                else:
 56                    camera_type = camera.get("camera_type", "")
 57                    url, _ = create_url(country, cid, camera_type)
 58                    media_type = (
 59                        "video" if camera_type in ["vid", "asfa_vid"] else "image"

 61

592    """

594
595    Args:
596        args (argparse.Namespace): The CLI arguments.

618        return
619
620    # Generate HTML
621    html_content = generate_html(cameras, args.interval, country)
622
623    # Save HTML file
624    if not args.output_file:
625        try:
626            args.output_file = f"Cameras_{COUNTRY_MAP[country]}.html"
627        except KeyError:
628            print(f"Warning: Unknown country code '{country}', using default filename")
629            args.output_file = "Cameras.html"

632    try:
633        if not output_path.parent.exists():
634            Path(output_path.parent).mkdir(parents=True, exist_ok=True)
635        output_path.write_text(html_content, encoding="utf-8")
636        print(f"HTML slideshow created: {output_path}")
637    except Exception as e:
638        print(f"Error writing HTML file: {e}")
